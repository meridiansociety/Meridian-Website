# ───────────────────────────────────────────────────────────────
# 1. ALLOW ALL MAJOR SEARCH ENGINE CRAWLERS
#    Full access to the entire site for all well-behaved bots.
#    This is the correct default for a public-facing site with
#    no private or authenticated content.
# ───────────────────────────────────────────────────────────────
User-agent: *
Allow: /

# ───────────────────────────────────────────────────────────────
# 2. GOOGLEBOT — EXPLICIT PERMISSIONS
#    Googlebot is the primary crawler. Explicit Allow rules
#    ensure Google indexes the homepage and all static assets
#    (fonts, images, JS, CSS) needed to render the page.
#    Google uses rendered HTML for indexing, so blocking assets
#    would prevent it from seeing your full content.
# ───────────────────────────────────────────────────────────────
User-agent: Googlebot
Allow: /
Allow: /favicon.svg
Allow: /favicon-32x32.png
Allow: /favicon-16x16.png
Allow: /apple-touch-icon.png
Allow: /og-image.png
Allow: /site.webmanifest

# ───────────────────────────────────────────────────────────────
# 3. GOOGLEBOT-IMAGE — ALLOW OG IMAGE FOR RICH RESULTS
#    Explicitly permits Google's image crawler to index your
#    Open Graph image, which is used in rich search results
#    and Google Discover cards.
# ───────────────────────────────────────────────────────────────
User-agent: Googlebot-Image
Allow: /og-image.png

# ───────────────────────────────────────────────────────────────
# 4. BINGBOT — MICROSOFT SEARCH
#    Bing powers Microsoft Edge, Cortana, and DuckDuckGo
#    results. Full access ensures coverage across all
#    major search platforms.
# ───────────────────────────────────────────────────────────────
User-agent: Bingbot
Allow: /

# ───────────────────────────────────────────────────────────────
# 5. BLOCK AI TRAINING CRAWLERS
#    These bots scrape content to train large language models.
#    Blocking them protects your original content from being
#    used as training data without consent, while having no
#    effect on your search engine rankings.
# ───────────────────────────────────────────────────────────────
User-agent: GPTBot
Disallow: /

User-agent: ChatGPT-User
Disallow: /

User-agent: CCBot
Disallow: /

User-agent: anthropic-ai
Disallow: /

User-agent: Claude-Web
Disallow: /

User-agent: cohere-ai
Disallow: /

Disallow: /

User-agent: FacebookBot
Disallow: /

# ───────────────────────────────────────────────────────────────
# 6. BLOCK AGGRESSIVE SEO SCRAPERS
#    These are commercial SEO audit tools that crawl sites
#    aggressively. Blocking them reduces unnecessary server
#    load without affecting your actual search rankings.
# ───────────────────────────────────────────────────────────────
User-agent: AhrefsBot
Disallow: /

User-agent: SemrushBot
Disallow: /

User-agent: MJ12bot
Disallow: /

Disallow: /
